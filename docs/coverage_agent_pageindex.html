24def _extract_text(payload: Any) -> str:
25 if payload is None:
26 return ""
27 if isinstance(payload, str):
28 return payload.strip()
29 if isinstance(payload, list):
30 parts = []
31 for item in payload:
32 if isinstance(item, str):
33 parts.append(item)
34 elif isinstance(item, dict) and item.get("type") == "text":
35 parts.append(str(item.get("text", "")))
36 return "\n".join(part.strip() for part in parts if part and part.strip()).strip()
37 return str(payload).strip()

40def _print_verbose_stream(agent: Any, agent_input: dict) -> None:
41 seen_messages: set[str] = set()
42
43 for chunk in agent.stream(agent_input, stream_mode="updates"):
44 if not isinstance(chunk, dict):
45 continue
46 for step_name, payload in chunk.items():
47 if not isinstance(payload, dict):
48 continue
49 messages = payload.get("messages") or []
50 for message in messages:
51 message_id = getattr(message, "id", None) or f"{step_name}:{type(message).__name__}:{len(seen_messages)}"
52 if message_id in seen_messages:
53 continue
54 seen_messages.add(message_id)
55
56 tool_calls = getattr(message, "tool_calls", None) or []
57 if tool_calls:
58 for call in tool_calls:
59 print(f"[tool call] {call.get('name')}({call.get('args')})")
60 continue
61
62 message_type = type(message).__name__
63 content = _extract_text(getattr(message, "content", ""))
64 if not content:
65 continue
66
67 if message_type == "ToolMessage":
68 preview = content if len(content) <= 1200 else content[:1200] + "..."
69 tool_name = getattr(message, "name", PAGEINDEX_TOOL_NAME)
70 print(f"[tool result] {tool_name}\n{preview}\n")
71 continue
72
73 if message_type == "AIMessage":
74 print(f"[{step_name}] {content}\n")
