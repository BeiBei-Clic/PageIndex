77 """Demonstrate try/except/else/finally.

82
83 Returns:

98def demo_nested_structures(data: list, threshold: int) -> list:
99 """Demonstrate nested loops inside conditionals.
100
101 Args:
102 data: List of integers to process
103 threshold: Controls whether processing happens
104
105 Returns:
106 Processed list of values
107 """
108 results = []
109
110 if threshold > 0:
111 for item in data:
112 count = 0
113 while count < threshold and item > 0:
114 results.append(item - count)
115 count += 1
116 else:
117 results = data.copy()
118
119 return results

161def _print_call_tree(call, depth: int = 0):
162 """Helper to print function call tree recursively.
163
164 Args:
165 call: FunctionCall object to print
166 depth: Current indentation depth
167 """
168 indent = " " * depth
169 args = ", ".join(f"{k}={v}" for k, v in call.arguments.items())
170 ret = f" -> {call.return_value}" if call.return_value else ""
171 print(f"{indent}{call.function_name}({args}){ret}")
172 for child in call.children:
173 _print_call_tree(child, depth + 1)
