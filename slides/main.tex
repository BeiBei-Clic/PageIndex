\section{研究背景}
\begin{frame}
\frametitle{研究背景：传统 RAG 的局限}
\begin{itemize}
    \item 传统向量检索依赖\textbf{语义相似度}而非真正的\textbf{相关性}
    \item 向量化需要 Chunking（分块），破坏文档原有结构
    \item 长文档场景下检索精度不足，缺乏可解释性
    \item 专业领域文档（金融、法律、学术）需要\textbf{多步推理}才能找到关键信息
\end{itemize}
\vspace{0.5cm}
\centering
\textbf{核心洞察：Similarity $\neq$ Relevance — 真正的检索需要推理能力}
\end{frame}

%------------------------------------------------

\section{PageIndex 框架}
\begin{frame}
\frametitle{PageIndex：无向量、基于推理的 RAG}
受 AlphaGo 启发，PageIndex 提出了一种全新的检索范式：
\begin{enumerate}
    \item \textbf{树结构索引}：将长文档转化为层次化的"目录"树
    \item \textbf{推理式检索}：利用 LLM 在树结构上进行推理式搜索
    \item \textbf{类人检索}：模拟人类专家浏览和提取知识的方式
\end{enumerate}
\vspace{0.3cm}
\begin{block}{核心特点}
\begin{itemize}
    \item 无需向量数据库（No Vector DB）
    \item 无需分块处理（No Chunking）
    \item 更好的可解释性与可追溯性
\end{itemize}
\end{block}
\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{两步工作流程}
\textbf{Step 1: 树结构索引生成}
\begin{itemize}
    \item 解析 PDF 文档，生成层次化树索引
    \item 每个节点包含标题、页码范围、摘要
\end{itemize}
\vspace{0.3cm}
\textbf{Step 2: 推理式树搜索}
\begin{itemize}
    \item LLM 在树结构上进行逐层推理
    \item 根据问题语义选择最相关的分支
    \item 返回精确的页码和段落引用
\end{itemize}
\end{frame}

\section{PathTracer：执行路径追踪}
\begin{frame}
\frametitle{PathTracer：代码执行路径追踪与覆盖率分析}
项目中的 \texttt{pathtracer} 模块，用于追踪 Python 程序的执行路径：
\begin{itemize}
    \item \textbf{分支追踪}：记录 if/else/for/while/try 等分支的执行情况
    \item \textbf{函数调用链}：记录函数调用的深度、参数和返回值
    \item \textbf{覆盖率报告}：生成 HTML/JSON/文本格式的覆盖率报告
\end{itemize}
\vspace{0.3cm}
\begin{block}{使用示例}
\tiny
\texttt{python -m pathtracer.cli agent\_pageindex.py --query "..." --doc-top-k 3}
\end{block}
\end{frame}
